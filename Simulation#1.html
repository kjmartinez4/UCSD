def monte(Temperature, CouplingParameter):
  """
  Returns plots of average magnetization per spin and heat capacity for a given
  temperature vector and J = 1
  """
  M_avg = np.zeros((len(CouplingParameter), N_T))   # magnetizations per spin [J/T]
  E_avg = np.zeros((len(CouplingParameter), N_T))   # energies per spin [J]
  Cv_avg = np.zeros((len(CouplingParameter), N_T))  # heat capacity per spin [k]
  for i1 in range(0,len(CouplingParameter)):        # Loops over the size of the given J vector
    for i2 in range(N_T):                           # Loops over the size of the given T vector
          print('Run:(' + str(i2 + 1) + ' of ' + str(N_T) + ')')
          
          # Initialization
          lattice = np.ones((ArrayDim, ArrayDim))                         # spin lattice
          spin_sum = ArrayDim ** 2                                        # spin sum of state
          E_sum = -2 * J[i1] * (ArrayDim ** 2)                            # energy of state [J]
          rand_flip = np.random.randint(ArrayDim, size = (2, Iterations)) # flipping indices
          num = np.random.random(Iterations)                              # acceptance criteria
          a = np.random.randint(0,ArrayDim ** 2-1)
          b = np.random.randint(0,ArrayDim ** 2-1)

          
          for j in range(Iterations):
              # Change in spin_sum and E_sum if a single spin is flipped from s state
              x = rand_flip[0, j]  # row index of spin to flip
              y = rand_flip[1, j]  # column index of spin to flip
              ds_sum = -2 * lattice[x, y]
              dE_sum = -J[i1] * ds_sum * (lattice[x - 1, y] + lattice[(x + 1) % ArrayDim, y] \
                                          + lattice[x, y - 1] + lattice[x, (y + 1) % ArrayDim])
      
              # Condition for accepting the new state
              alpha = math.exp(-dE_sum / T[i2])  # probability of acceptance 
              if num[j] <= alpha:                # if statement to accept or not
                  lattice[x, y] *= -1            # flip the spin
                  E_sum += dE_sum                # energy for new state
                  spin_sum += ds_sum             # spin sum for new state
                  
                  
              # Sum of |M| and E
              if j >= N_trunc:
                  E_avg[i1,i2] += E_sum
                  M_avg[i1,i2] += mu * abs(spin_sum)
                  
      
          # Ensemble absolute average per spin for magnetization and energy: |M/N| and E/N
          M_avg[i1,i2] /= (ArrayDim ** 2) * (Iterations - N_trunc)
          E_avg[i1,i2] /= (ArrayDim ** 2) * (Iterations - N_trunc)

  # Specific Heat: Cv/N       
  inc_T = T[1] - T[0]
  for i2 in range(1, N_T - 1):
      Cv_avg[i1,i2] = (E_avg[i1,i2 + 1] - E_avg[i1,i2 - 1]) / (2 * inc_T)

  # Plot
  print('Plotting M/N vs T...')

  plt.figure(0)
  for i1 in range(len(CouplingParameter)):
    plt.plot(T, M_avg[i1, :], '+', 
             label = 'J = 1 ')
  plt.title('Ensemble-average of Magnetization per Spin')
  plt.xlabel('T [J/$k_{B}$]')
  plt.ylabel('|M/N| [J/T]')
  plt.legend()
  plt.savefig('Plot P1 (1).jpg')
      
  plt.figure(1)
  for i1 in range(len(CouplingParameter)):
    plt.plot(T[1:-1], Cv_avg[i1, 1:-1], '+', 
              label = 'J = 1')
  plt.title('Heat Capacity per Spin')
  plt.xlabel('T [J/$k_{B}$]')
  plt.ylabel('$C_{V}$/N [$k_{B}$]')
  plt.legend()
  plt.savefig('Plot P1 (2).jpg')

def MonteCarloc(Temperature,CouplingParameter):

# Vector of ensemble-averages
  M_avg = np.zeros((N_J, N_T))   # |M/N|: magnetizations per spin [J/T]
  E_avg = np.zeros((N_J, N_T))   # E/N: energies per spin [J]
  Cv_avg = np.zeros((N_J, N_T))  # Cv/N: heat capacity per spin [k]
  for i1 in range(N_J):
      for i2 in range(N_T):
          print('Run: J = ' + str(round(J[i1], 2)) 
                + ' (' + str(i2 + 1) + ' of ' + str(N_T) + ')')
          
          # Initialization
          s = np.ones((ArrayDim, ArrayDim))                        # spin lattice
          s_sum = ArrayDim ** 2                                  # spin sum of state
          E_sum = -2 * J[i1] * (ArrayDim ** 2)                   # energy of state [J]
          idx = np.random.randint(ArrayDim, size = (2, Iterations))  # flipping indices
          p = np.random.random(Iterations)                         # acceptance criteria
          
          for j in range(Iterations):
              # Change in s_sum and E_sum if a single spin is flipped from s state
              k = idx[0, j]  # row index of spin to flip
              l = idx[1, j]  # column index of spin to flip
              ds_sum = -2 * s[k, l]
              dE_sum = -J[i1] * ds_sum * (s[k - 1, l] + s[(k + 1) % ArrayDim, l] \
                                          + s[k, l - 1] + s[k, (l + 1) % ArrayDim])
      
              # Condition for accepting the new state
              alpha = math.exp(-dE_sum / T[i2])  # probability of acceptance 
              if p[j] <= alpha:                  # accepting condition
                  s[k, l] *= -1                  # flip the spin
                  s_sum += ds_sum                # spin sum for new state
                  E_sum += dE_sum                # energy for new state
                  
              # Sum of |M| and E
              if j >= N_trunc:
                  M_avg[i1, i2] += mu * abs(s_sum)
                  E_avg[i1, i2] += E_sum
      
          # Average of |M/N| and E/N
          M_avg[i1, i2] /= (ArrayDim ** 2) * (Iterations - N_trunc)
          E_avg[i1, i2] /= (ArrayDim ** 2) * (Iterations - N_trunc)

  # Timer end    
  t_end = time.time()                          
  print(f"Total runtime: {t_end - t_start}s")  # calculate the total runtime

  # Calculation of Cv/N
  inc_T = T[1] - T[0]
  for i1 in range(N_J):
      for i2 in range(1, N_T - 1):
          Cv_avg[i1, i2] = (E_avg[i1, i2 + 1] - E_avg[i1, i2 - 1]) / (2 * inc_T)

  # Plot
  print('Plotting Magnetization(M) and Specific Heat(Cv) per Spin versus Temperature(T) for different values of J...')
  color_base = [0.05, 0.20, 0.35, 0.50, 0.65, 0.80, 0.95]
